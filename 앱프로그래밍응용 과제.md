### 2. 정사각형의 길이를 입력하고 사각형을 출력
```dart
var result = '';
void main() {
  var n = 10;

  for (var y = 0; y < n; y++) {
    for (var x = 0; x < n; x++) {
      if (pattern1(n, x, y)) {
        result += '= ';
      } else {
        result += '  ';
      }
    }
    result += '\n';
  }
  print(result);
}

// 꽉 찬 사각형
bool pattern1(int n, int x, int y) {
  return true;
}

// 테두리 사각형
bool pattern2(int n, int x, int y) {
  var condition = y == 0;
  condition |= y == (n - 1);
  condition |= x == 0;
  condition |= x == (n - 1);
  return condition;
}

// \표시
bool pattern3(int n, int x, int y) {
  var condition = y == 0;
  condition |= y == (n - 1);
  condition |= x == 0;
  condition |= x == (n - 1);
  condition |= x == y;
  return condition;
}

// /표시
bool pattern4(int n, int x, int y) {
  var condition = y == 0;
  condition |= y == (n - 1);
  condition |= x == 0;
  condition |= x == (n - 1);
  condition |= x + y == n - 1;
  return condition;
}

// X표시
bool pattern5(int n, int x, int y) {
  var condition = y == 0;
  condition |= y == (n - 1);
  condition |= x == 0;
  condition |= x == (n - 1);
  condition |= x == y || x + y == n - 1;
  return condition;
}
```
### 3. 년/월/일을 입력하면 요일을 출력하라
```dart
void main() {
  // DateTime 클래스를 사용하여 date 변수에 DateTime(연도, 월, 일) 형태로 날짜를 지정
  DateTime date = DateTime(2026, 5, 12);
  // 요일 목록을 리스트로 저장
  List<String> weekdays = ['월요일', '화요일', '수요일', '목요일', '금요일', '토요일', '일요일'];

  // date.weekday를 통해 요일을 숫자로 변환 후 weekdays의 리스트 목록의 요일과 대조하여 요일 표시
  print('2026년 5월 12일은 ${weekdays[date.weekday - 1]}입니다.');
}
```

# ⚙ 앱프로그래밍 응용 4주차 과제
### 현재 시각을 표시하는 앱
```dart
import 'dart:async';
import 'package:flutter/material.dart';
import 'package:intl/intl.dart';

void main() => runApp(MyApp());

class MyApp extends StatelessWidget {
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      title: '현재 시각을 표시하는 앱',
      theme: ThemeData(
        colorScheme: ColorScheme.fromSeed(seedColor: Colors.deepPurple),
      ),
      home: MyHomePage(title: '현재 시각'),
    );
  }
}

class MyHomePage extends StatefulWidget {
  const MyHomePage({super.key, required this.title});
  final String title;

  @override
  State<MyHomePage> createState() => _MyHomePageState();
}

class _MyHomePageState extends State<MyHomePage> {
  String _currentTime = '';
  late Timer _timer;

  @override
  void initState() {
    super.initState();
    _updateTime();
    _timer = Timer.periodic(Duration(seconds: 1), (timer) => _updateTime());
  }

  // 현재 시간을 업데이트하는 함수
  void _updateTime() {
    setState(() {
      _currentTime = DateFormat('yyyy-MM-dd\nHH:mm:ss').format(DateTime.now());
    });
  }

  @override
  void dispose() {
    _timer.cancel(); // 타이머 해제
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        backgroundColor: Theme.of(context).colorScheme.inversePrimary,
        title: Text(widget.title),
      ),
      body: Center(
        child: Column(
          mainAxisAlignment: MainAxisAlignment.center,
          children: <Widget>[
            Text(
              _currentTime,
              style: Theme.of(context).textTheme.headlineMedium,
            ),
          ],
        ),
      ),
    );
  }
}
```

### 계산기 UI 구성
```dart
import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      home: const CalculatorSimpleUI(),
      debugShowCheckedModeBanner: false,
    );
  }
}

class CalculatorSimpleUI extends StatelessWidget {
  const CalculatorSimpleUI({super.key});

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      backgroundColor: Colors.cyan, // 전체 배경색
      body: Column(
        children: [
          // 결과창
          Expanded(
            flex: 2,
            child: Container(
              color: Colors.black,
              alignment: Alignment.bottomRight,
              padding: const EdgeInsets.all(16),
              child: const Text(
                '0',
                style: TextStyle(
                  color: Colors.white,
                  fontSize: 48,
                ),
              ),
            ),
          ),

          // 버튼들
          Expanded(
            flex: 5,
            child: Column(
              children: [
                buildRow(['7', '8', '9', '÷']),
                buildRow(['4', '5', '6', '×']),
                buildRow(['1', '2', '3', '-']),
                buildRow(['0', '.', '=', '+']),
              ],
            ),
          ),
        ],
      ),
    );
  }

  Widget buildRow(List<String> labels) {
    return Expanded(
      child: Row(
        children: labels.map((label) {
          return Expanded(
            child: Container(
              margin: const EdgeInsets.all(1),
              color: label == '=' ? Colors.lightBlueAccent : Colors.grey[850],
              alignment: Alignment.center,
              child: Text(
                label,
                style: TextStyle(
                  fontSize: 40,
                  color: label == '=' ? Colors.black : Colors.white,
                ),
              ),
            ),
          );
        }).toList(),
      ),
    );
  }
}
```
