### 2. 🛠정사각형의 길이를 입력하고 사각형을 출력
```dart
var result = '';
void main() {
  var n = 10;

  for (var y = 0; y < n; y++) {
    for (var x = 0; x < n; x++) {
      if (pattern4(n, x, y)) {
        result += '= ';
      } else {
        result += '  ';
      }
    }
    result += '\n';
  }
  print(result);
}

bool pattern1(int n, int x, int y) {
  var condition = y == 0;
  condition |= y == (n - 1);
  condition |= x == 0;
  condition |= x == (n - 1);
  return condition;
}

bool pattern2(int n, int x, int y) {
  var condition = y == 0;
  condition |= y == (n - 1);
  condition |= x == 0;
  condition |= x == (n - 1);
  condition |= x == y;
  return condition;
}

bool pattern3(int n, int x, int y) {
  var condition = y == 0;
  condition |= y == (n - 1);
  condition |= x == 0;
  condition |= x == (n - 1);
  condition |= x + y == n - 1;
  return condition;
}

bool pattern4(int n, int x, int y) {
  var condition = y == 0;
  condition |= y == (n - 1);
  condition |= x == 0;
  condition |= x == (n - 1);
  condition |= x == y || x + y == n - 1;
  return condition;
}
```
