### 2. 정사각형의 길이를 입력하고 사각형을 출력
```dart
var result = '';
void main() {
  var n = 10;

  for (var y = 0; y < n; y++) {
    for (var x = 0; x < n; x++) {
      if (pattern1(n, x, y)) {
        result += '= ';
      } else {
        result += '  ';
      }
    }
    result += '\n';
  }
  print(result);
}

// 꽉 찬 사각형
bool pattern1(int n, int x, int y) {
  return true;
}

// 테두리 사각형
bool pattern2(int n, int x, int y) {
  var condition = y == 0;
  condition |= y == (n - 1);
  condition |= x == 0;
  condition |= x == (n - 1);
  return condition;
}

// \표시
bool pattern3(int n, int x, int y) {
  var condition = y == 0;
  condition |= y == (n - 1);
  condition |= x == 0;
  condition |= x == (n - 1);
  condition |= x == y;
  return condition;
}

// /표시
bool pattern4(int n, int x, int y) {
  var condition = y == 0;
  condition |= y == (n - 1);
  condition |= x == 0;
  condition |= x == (n - 1);
  condition |= x + y == n - 1;
  return condition;
}

// X표시
bool pattern5(int n, int x, int y) {
  var condition = y == 0;
  condition |= y == (n - 1);
  condition |= x == 0;
  condition |= x == (n - 1);
  condition |= x == y || x + y == n - 1;
  return condition;
}
```
### 3. 년/월/일을 입력하면 요일을 출력하라
```dart
void main() {
  DateTime date = DateTime(2026, 5, 12);
  List<String> weekdays = ['월요일', '화요일', '수요일', '목요일', '금요일', '토요일', '일요일'];

  print('2026년 5월 12일은 ${weekdays[date.weekday - 1]}입니다.');
}
```
